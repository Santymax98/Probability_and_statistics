\textbf{🟢 1. Permutaciones completas sin repetición}
\textbf{📌 ¿Qué es?}
\text{Ordenamiento de todos los elementos de un conjunto, sin repeticiones.}
\textbf{📐 Fórmula:}
nPn = n!
\textbf{🧠 Interpretación}
\text{Número de formas de ordenar} \ n \ \text{elementos distintos.}
\textbf{🧪 Ejemplo clásico}
\text{Ordenar 5 libros en una repisa:}
5! = 120

🧩 Detalle de cada caso

\textbf{🟢 2. Permutaciones parciales sin repetición}
\textbf{📌 ¿Qué es?}
\text{Seleccionar y ordenar solo} \ r \ \text{elementos de un total de} \ n, \ \text{sin repetir.}
\textbf{📐 Fórmula:}
nPr = \frac{n!}{(n-r)!}
\textbf{🧠 Interpretación}
\text{Número de formas de formar listas ordenadas de longitud} \ r \ \text{con elementos distintos.}
\textbf{🧪 Ejemplo clásico}
\text{Asignar oro, plata y bronce a 10 atletas:}
10P3 = \frac{10!}{(10-3)!}=720

🧮 Variante: Soluciones enteras positivas

\textbf{💡 Idea:}
\text{Si en lugar de} \ x_i \geq 0, \ \text{se pide} \ x_i \geq 1, \ \text{entonces hay que restar 1 a cada variable}
\text{de antemano. Esto equivale a:}
x_1 + x_2 + \dots + x_k = n \quad \text{con } x_i \geq 1 \quad \Longleftrightarrow \quad \binom{n - 1}{k - 1}

\textbf{🔄 Conexión con combinaciones con repetición}
\text{Este tipo de conteo es una aplicación directa de las combinaciones con repetición:}
\binom{n + k - 1}{k - 1} = \binom{n+k-1}{n}
\text{😕 La diferencia es interpretativa:}
\text{En combinaciones: se elige un multiconjunto.}
\text{En soluciones enteras: se reparte un total entre variables.}
